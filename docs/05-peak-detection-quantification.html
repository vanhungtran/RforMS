#' Filter peaks by minimum distance threshold
#' 
#' Removes closely spaced peaks, keeping only the highest peak in each group
#' 
#' @param peak_data data.frame with peak information (must have 'index' and 'height' columns)
#' @param min_distance numeric, minimum index distance between peaks
#' @param verbose logical, whether to print filtering messages
#' @return filtered peak data.frame
filter_peaks_by_distance <- function(peak_data, min_distance = 8, verbose = FALSE) {
  # Input validation
  if (min_distance < 1) {
    stop("Minimum distance must be greater than 0")
  }
  
  if (!inherits(peak_data, "data.frame") || 
      !all(c("index", "height") %in% names(peak_data))) {
    stop("Input must be a data.frame with 'index' and 'height' columns")
  }
  
  if (nrow(peak_data) == 0) {
    return(peak_data)
  }
  
  # Calculate distances between consecutive peaks
  peak_indices <- peak_data$index
  peak_heights <- peak_data$height
  index_differences <- diff(peak_indices)
  
  # Identify peaks that are too close together
  too_close <- index_differences < min_distance
  
  if (!any(too_close)) {
    if (verbose) {
      message(sprintf("No peaks removed (all peaks separated by >= %d indices)", 
                      min_distance))
    }
    return(peak_data)
  }
  
  # Find groups of adjacent peaks that are too close
  close_indices <- which(too_close)
  peak_groups <- split(close_indices, 
                       cumsum(c(1, diff(close_indices) != 1)))
  
  # For each group, keep only the peak with maximum height
  peaks_to_remove <- integer(0)
  for (group in peak_groups) {
    # Include the peak before the group
    group_start <- if (group[1] > 1) group[1] else 1
    group_end <- group[length(group)] + 1
    
    # Get all peak indices in this group
    group_peak_indices <- group_start:group_end
    
    # Find the peak with maximum height in this group
    group_heights <- peak_heights[group_peak_indices]
    max_height_idx <- group_peak_indices[which.max(group_heights)]
    
    # Mark all others for removal
    peaks_to_remove <- c(peaks_to_remove, 
                         setdiff(group_peak_indices, max_height_idx))
  }
  
  # Remove the selected peaks
  filtered_peaks <- peak_data[-peaks_to_remove, , drop = FALSE]
  num_removed <- length(peaks_to_remove)
  
  if (verbose && num_removed > 0) {
    message(sprintf("%d peak%s removed using distance threshold (d=%d)", 
                    num_removed, 
                    ifelse(num_removed == 1, "", "s"), 
                    min_distance))
  }
  
  return(filtered_peaks)
}

#' Detect peaks in spectral data with configurable parameters
#' 
#' Identifies local maxima in spectral data with neighbor validation,
#' minimum height filtering, and distance-based filtering
#' 
#' @param spectral_data numeric vector of intensity values (can be named with m/z values)
#' @param neighbor_count integer, number of neighbors to check on each side (default: 3)
#' @param min_height numeric, optional minimum peak height threshold
#' @param min_distance numeric, optional minimum index distance between peaks
#' @param peak_type character, label for peak type (default: "peak")
#' @param verbose logical, whether to print processing messages
#' @return data.frame with detected peaks (index, peak position, height, type)
detect_spectral_peaks <- function(spectral_data, 
                                   neighbor_count = 3,
                                   min_height = NULL,
                                   min_distance = NULL,
                                   peak_type = "peak",
                                   verbose = FALSE) {
  # Input validation
  if (!is.numeric(spectral_data) || length(spectral_data) < 3) {
    stop("Spectral data must be a numeric vector with at least 3 points")
  }
  
  if (neighbor_count < 1) {
    stop("Neighbor count must be at least 1")
  }
  
  data_length <- length(spectral_data)
  data_names <- names(spectral_data)
  
  # Step 1: Find all local maxima (simple peak detection)
  # Compare each point with its immediate neighbors
  candidate_indices <- seq.int(2, data_length - 1)
  is_local_maximum <- spectral_data[candidate_indices] > 
                      spectral_data[candidate_indices - 1] &
                      spectral_data[candidate_indices] > 
                      spectral_data[candidate_indices + 1]
  
  peak_candidates <- candidate_indices[is_local_maximum]
  
  # Step 2: Filter by minimum height if specified
  if (!is.null(min_height)) {
    peak_heights <- spectral_data[peak_candidates]
    above_threshold <- peak_heights >= min_height
    peak_candidates <- peak_candidates[above_threshold]
    
    if (verbose) {
      num_filtered <- sum(!above_threshold)
      if (num_filtered > 0) {
        message(sprintf("%d peak%s below minimum height threshold (%.2f)", 
                        num_filtered,
                        ifelse(num_filtered == 1, "", "s"),
                        min_height))
      }
    }
  }
  
  # Step 3: Validate peaks using neighbor checking
  if (neighbor_count > 1 && length(peak_candidates) > 0) {
    # Remove peaks too close to boundaries
    valid_range <- (neighbor_count + 1):(data_length - neighbor_count)
    peak_candidates <- peak_candidates[peak_candidates %in% valid_range]
    
    if (length(peak_candidates) > 0) {
      # Create validation matrix: check n neighbors on each side
      validation_results <- matrix(FALSE, 
                                   nrow = length(peak_candidates), 
                                   ncol = neighbor_count)
      
      for (neighbor_idx in seq_len(neighbor_count)) {
        offset <- neighbor_idx
        # Check that peak is higher than neighbors at this offset
        left_neighbors <- spectral_data[peak_candidates - offset]
        right_neighbors <- spectral_data[peak_candidates + offset]
        peak_values <- spectral_data[peak_candidates]
        
        validation_results[, neighbor_idx] <- 
          peak_values > left_neighbors & peak_values > right_neighbors
      }
      
      # Keep only peaks that pass all neighbor checks
      all_checks_passed <- rowSums(validation_results) == neighbor_count
      validated_peaks <- peak_candidates[all_checks_passed]
    } else {
      validated_peaks <- integer(0)
    }
  } else {
    validated_peaks <- peak_candidates
  }
  
  # Step 4: Create results data frame
  if (length(validated_peaks) > 0) {
    peak_positions <- if (!is.null(data_names)) {
      data_names[validated_peaks]
    } else {
      as.character(validated_peaks)
    }
    
    peak_results <- data.frame(
      index = validated_peaks,
      position = peak_positions,
      height = as.numeric(spectral_data[validated_peaks]),
      type = peak_type,
      stringsAsFactors = FALSE
    )
    
    # Add class attribute for method dispatch
    class(peak_results) <- c("spectral_peaks", "data.frame")
    
    # Step 5: Apply distance filtering if specified
    if (!is.null(min_distance) && min_distance > 0) {
      peak_results <- filter_peaks_by_distance(peak_results, 
                                                min_distance, 
                                                verbose)
    }
    
    return(peak_results)
  } else {
    # Return empty data frame with correct structure
    empty_result <- data.frame(
      index = integer(0),
      position = character(0),
      height = numeric(0),
      type = character(0),
      stringsAsFactors = FALSE
    )
    class(empty_result) <- c("spectral_peaks", "data.frame")
    return(empty_result)
  }
}

6.7.3 Peak Slope Analysis

Code
#' Calculate peak slope angles and asymmetry metrics
#' 
#' Analyzes the geometric properties of peaks by calculating
#' slope angles on both sides and measuring peak asymmetry
#' 
#' @param spectral_data numeric vector of intensity values
#' @param peak_data data.frame with peak information (must have 'index' column)
#' @param neighbor_offset integer, distance from peak to measure slope (default: 3)
#' @param use_absolute logical, whether to use absolute angles (default: TRUE)
#' @param decimal_places integer, number of decimal places for rounding (default: 2)
#' @return data.frame with peak data and slope metrics
calculate_peak_slope_metrics <- function(spectral_data,
                                         peak_data,
                                         neighbor_offset = 3,
                                         use_absolute = TRUE,
                                         decimal_places = 2) {
  # Input validation
  if (!inherits(peak_data, "data.frame") || !"index" %in% names(peak_data)) {
    stop("peak_data must be a data.frame with an 'index' column")
  }
  
  if (nrow(peak_data) == 0) {
    return(peak_data)
  }
  
  data_length <- length(spectral_data)
  peak_indices <- peak_data$index
  
  # Initialize result columns
  left_angles <- numeric(nrow(peak_data))
  right_angles <- numeric(nrow(peak_data))
  
  # Calculate slope angles for each peak
  for (i in seq_len(nrow(peak_data))) {
    peak_idx <- peak_indices[i]
    
    # Check if peak is far enough from boundaries
    if (peak_idx <= neighbor_offset || peak_idx > (data_length - neighbor_offset)) {
      left_angles[i] <- NA
      right_angles[i] <- NA
      next
    }
    
    # Get coordinates for slope calculation
    peak_x <- peak_idx
    peak_y <- spectral_data[peak_idx]
    
    # Left side slope
    left_x <- peak_idx - neighbor_offset
    left_y <- spectral_data[left_x]
    left_slope_radians <- atan2(left_y - peak_y, left_x - peak_x)
    left_angles[i] <- left_slope_radians * 180 / pi
    
    # Right side slope
    right_x <- peak_idx + neighbor_offset
    right_y <- spectral_data[right_x]
    right_slope_radians <- atan2(right_y - peak_y, right_x - peak_x)
    right_angles[i] <- right_slope_radians * 180 / pi
  }
  
  # Convert to absolute values if requested
  if (use_absolute) {
    left_angles <- abs(left_angles)
    right_angles <- abs(right_angles)
  }
  
  # Calculate mean slope angle
  mean_angles <- rowMeans(cbind(left_angles, right_angles), na.rm = TRUE)
  
  # Calculate asymmetry (percent difference between left and right slopes)
  # Higher values indicate more asymmetric peaks
  angle_differences <- abs(left_angles - right_angles)
  asymmetry_percent <- ifelse(mean_angles > 0,
                              (angle_differences / mean_angles) * 100,
                              NA)
  
  # Combine results
  result <- cbind(
    peak_data,
    data.frame(
      left_slope_angle = round(left_angles, decimal_places),
      right_slope_angle = round(right_angles, decimal_places),
      mean_slope_angle = round(mean_angles, decimal_places),
      asymmetry_percent = round(asymmetry_percent, decimal_places),
      stringsAsFactors = FALSE
    )
  )
  
  rownames(result) <- NULL
  return(result)
}

6.7.4 Trough Detection

Code
#' Detect troughs (local minima) in spectral data
#' 
#' Identifies local minima by inverting the data and applying
#' peak detection algorithms
#' 
#' @param spectral_data numeric vector of intensity values
#' @param neighbor_count integer, number of neighbors to check (default: 3)
#' @param min_depth numeric, optional minimum trough depth
#' @param min_distance numeric, optional minimum distance between troughs
#' @param rename_column logical, whether to rename 'position' to 'trough' (default: TRUE)
#' @return data.frame with detected troughs
detect_spectral_troughs <- function(spectral_data,
                                     neighbor_count = 3,
                                     min_depth = NULL,
                                     min_distance = NULL,
                                     rename_column = TRUE) {
  # Invert the data: minima become maxima
  inverted_data <- -spectral_data
  
  # Adjust minimum depth threshold if provided
  adjusted_min_height <- if (!is.null(min_depth)) {
    -min_depth
  } else {
    NULL
  }
  
  # Detect peaks in inverted data (these are troughs in original)
  trough_results <- detect_spectral_peaks(
    spectral_data = inverted_data,
    neighbor_count = neighbor_count,
    min_height = adjusted_min_height,
    min_distance = min_distance,
    peak_type = "trough",
    verbose = FALSE
  )
  
  # Replace heights with original (non-inverted) values
  if (nrow(trough_results) > 0) {
    trough_results$height <- spectral_data[trough_results$index]
    
    # Optionally rename position column
    if (rename_column && "position" %in% names(trough_results)) {
      names(trough_results)[names(trough_results) == "position"] <- "trough"
    }
  }
  
  return(trough_results)
}

#' Calculate polygon area using the shoelace formula
#' 
#' Computes the area of a polygon defined by x,y coordinates
#' using the shoelace (surveyor's) formula. Useful for calculating
#' peak areas in spectral data.
#' 
#' @param x_coords numeric vector of x-coordinates
#' @param y_coords numeric vector of y-coordinates
#' @return numeric, area of the polygon
calculate_polygon_area <- function(x_coords, y_coords) {
  # Input validation
  if (length(x_coords) != length(y_coords)) {
    stop("x and y coordinate vectors must have the same length")
  }
  
  if (any(is.na(x_coords)) || any(is.na(y_coords))) {
    return(NA_real_)
  }
  
  if (length(x_coords) < 3) {
    return(0)
  }
  
  n_points <- length(x_coords)
  
  # Shoelace formula: area = 0.5 * |sum(x_i * y_{i+1} - x_{i+1} * y_i)|
  # Create circular index for wrapping around
  next_index <- c(seq.int(2, n_points), 1)
  
  # Calculate cross products
  cross_products <- x_coords * y_coords[next_index] - 
                    x_coords[next_index] * y_coords
  
  # Sum and take absolute value, then halve
  polygon_area <- abs(sum(cross_products)) * 0.5
  
  return(polygon_area)
}

#' Refine peak boundaries using linear regression baseline correction
#' 
#' Uses linear regression to identify and remove baseline contributions
#' from peak area calculations, improving accuracy
#' 
#' @param peak_regions list of data.frames, each containing x, y coordinates for a peak region
#' @param rounding_digits integer, number of decimal places for comparison (default: 8)
#' @return list of refined peak region data.frames
refine_peak_boundaries <- function(peak_regions, rounding_digits = 8) {
  if (!is.list(peak_regions) || length(peak_regions) == 0) {
    return(peak_regions)
  }
  
  refined_regions <- vector("list", length(peak_regions))
  
  for (i in seq_along(peak_regions)) {
    region_data <- peak_regions[[i]]
    
    # Skip if not a valid data frame
    if (!inherits(region_data, "data.frame") || nrow(region_data) < 2) {
      refined_regions[[i]] <- region_data
      next
    }
    
    # Extract coordinates
    x_coords <- region_data[[1]]  # First column (x_in or x_p)
    y_coords <- region_data[[3]]  # Third column (y)
    
    # Fit linear regression using first and last points (baseline)
    n_points <- nrow(region_data)
    baseline_points <- data.frame(
      x = c(x_coords[1], x_coords[n_points]),
      y = c(y_coords[1], y_coords[n_points])
    )
    
    baseline_model <- tryCatch(
      lm(y ~ x, data = baseline_points),
      error = function(e) NULL
    )
    
    if (is.null(baseline_model)) {
      refined_regions[[i]] <- region_data
      next
    }
    
    # Predict baseline values for all points
    predicted_baseline <- predict(baseline_model, 
                                   newdata = data.frame(x = x_coords))
    
    # Calculate difference between actual and predicted (baseline-corrected)
    baseline_corrected <- round(y_coords, rounding_digits) - 
                          round(predicted_baseline, rounding_digits)
    
    # Split region into two halves
    midpoint <- ceiling(n_points / 2)
    first_half <- seq_len(midpoint)
    second_half <- seq(midpoint + 1, n_points)
    
    # Find negative values (below baseline) in each half
    negative_first <- which(baseline_corrected[first_half] < 0)
    negative_second <- which(baseline_corrected[second_half] < 0)
    
    # Adjust indices for second half
    if (length(negative_second) > 0) {
      negative_second <- negative_second + midpoint
    }
    
    # Remove points below baseline, keeping the peak region
    indices_to_keep <- seq_len(n_points)
    
    # Remove from end (second half) first
    if (length(negative_second) > 0) {
      first_negative_end <- min(negative_second)
      indices_to_keep <- indices_to_keep[indices_to_keep < first_negative_end]
    }
    
    # Remove from beginning (first half)
    if (length(negative_first) > 0) {
      last_negative_start <- max(negative_first)
      indices_to_keep <- indices_to_keep[indices_to_keep > last_negative_start]
    }
    
    # Create refined region
    if (length(indices_to_keep) >= 2) {
      refined_regions[[i]] <- region_data[indices_to_keep, , drop = FALSE]
    } else {
      refined_regions[[i]] <- region_data
    }
  }
  
  return(refined_regions)
}

#' Calculate peak areas for specified peaks in spectral data matrix
#' 
#' Detects peaks and troughs, then calculates areas for user-specified peaks
#' using the region between adjacent troughs
#' 
#' @param spectral_matrix matrix or data.frame, columns are spectra, rows are m/z values
#' @param target_peak character or numeric, peak identifier to calculate area for
#' @param neighbor_count integer, number of neighbors for peak detection (default: 3)
#' @param coordinate_type character, use "index" or "position" for area calculation (default: "index")
#' @param refine_baseline logical, whether to refine boundaries using regression (default: TRUE)
#' @param rounding_precision integer, decimal places for baseline refinement (default: 8)
#' @param ... additional arguments passed to peak detection function
#' @return list with peak coordinates, calculated areas, and detection results
calculate_peak_areas <- function(spectral_matrix,
                                 target_peak,
                                 neighbor_count = 3,
                                 coordinate_type = "index",
                                 refine_baseline = TRUE,
                                 rounding_precision = 8,
                                 ...) {
  # Input validation
  if (!is.matrix(spectral_matrix) && !is.data.frame(spectral_matrix)) {
    stop("spectral_matrix must be a matrix or data.frame")
  }
  
  # Calculate spectral resolution (m/z step size)
  row_names <- rownames(spectral_matrix)
  if (is.null(row_names) || length(row_names) < 2) {
    stop("spectral_matrix must have row names representing m/z values")
  }
  
  spectral_resolution <- abs(as.numeric(row_names[1]) - as.numeric(row_names[2]))
  n_spectra <- ncol(spectral_matrix)
  
  # Detect peaks and troughs for each spectrum
  detected_peaks <- apply(spectral_matrix, 2, function(spectrum) {
    detect_spectral_peaks(spectrum, neighbor_count = neighbor_count, 
                         verbose = FALSE, ...)
  }, simplify = FALSE)
  
  detected_troughs <- apply(spectral_matrix, 2, function(spectrum) {
    detect_spectral_troughs(spectrum, neighbor_count = 1, 
                           rename_column = FALSE)
  }, simplify = FALSE)
  
  # Combine peaks and troughs, sort by index
  combined_features <- mapply(function(peaks, troughs) {
    combined <- rbind(peaks, troughs)
    combined <- combined[order(combined$index), ]
    rownames(combined) <- NULL
    return(combined)
  }, detected_peaks, detected_troughs, SIMPLIFY = FALSE)
  
  # Find target peak in each spectrum
  peak_regions <- vector("list", n_spectra)
  calculated_areas <- vector("list", n_spectra)
  
  for (spec_idx in seq_len(n_spectra)) {
    features <- combined_features[[spec_idx]]
    
    # Find target peak
    peak_matches <- grep(as.character(target_peak), features$position, 
                        value = FALSE, fixed = TRUE)
    peak_rows <- features[peak_matches, , drop = FALSE]
    peak_rows <- peak_rows[peak_rows$type == "peak", , drop = FALSE]
    
    if (nrow(peak_rows) == 0) {
      peak_regions[[spec_idx]] <- NULL
      calculated_areas[[spec_idx]] <- NA_real_
      next
    }
    
    # Use first matching peak
    target_peak_idx <- peak_rows$index[1]
    feature_indices <- features$index
    
    # Find adjacent troughs
    peak_position <- which(feature_indices == target_peak_idx)
    left_trough_idx <- if (peak_position > 1) {
      features$index[peak_position - 1]
    } else {
      1
    }
    right_trough_idx <- if (peak_position < length(feature_indices)) {
      features$index[peak_position + 1]
    } else {
      nrow(spectral_matrix)
    }
    
    # Extract region coordinates
    region_indices <- left_trough_idx:right_trough_idx
    region_mz <- as.numeric(row_names[region_indices])
    region_intensity <- spectral_matrix[region_indices, spec_idx]
    
    # Create region data frame
    region_data <- data.frame(
      index = region_indices,
      mz = region_mz,
      intensity = region_intensity,
      stringsAsFactors = FALSE
    )
    
    peak_regions[[spec_idx]] <- region_data
    
    # Refine boundaries if requested
    if (refine_baseline) {
      refined_list <- refine_peak_boundaries(list(region_data), 
                                             rounding_precision)
      region_data <- refined_list[[1]]
    }
    
    # Calculate area using appropriate coordinates
    if (coordinate_type == "index") {
      x_coords <- region_data$index
    } else {
      x_coords <- region_data$mz
    }
    y_coords <- region_data$intensity
    
    calculated_areas[[spec_idx]] <- calculate_polygon_area(x_coords, y_coords)
  }
  
  return(list(
    peak_regions = peak_regions,
    areas = calculated_areas,
    detected_peaks = detected_peaks,
    detected_troughs = detected_troughs
  ))
}

  region_data <- data.frame(

#' Calculate descriptive statistics for peak areas
#' 
#' Computes mean, median, standard deviation, and standard error
#' with optional outlier removal
#' 
#' @param area_values numeric vector or list, peak area values
#' @param remove_outliers logical, whether to remove outliers (default: TRUE)
#' @param iqr_multiplier numeric, IQR multiplier for outlier detection (default: 1.5)
#' @return list with area values and statistics
calculate_area_statistics <- function(area_values,
                                      remove_outliers = TRUE,
                                      iqr_multiplier = 1.5) {
  # Convert to vector if needed
  if (is.list(area_values)) {
    area_values <- unlist(area_values)
  }
  
  # Remove infinite values
  area_values <- area_values[is.finite(area_values)]
  
  if (length(area_values) == 0) {
    return(list(
      values = numeric(0),
      mean = NA_real_,
      median = NA_real_,
      std_deviation = NA_real_,
      std_error = NA_real_
    ))
  }
  
  # Remove outliers using IQR method
  if (remove_outliers && length(area_values) > 4) {
    quartiles <- quantile(area_values, probs = c(0.25, 0.75), na.rm = TRUE)
    iqr <- IQR(area_values, na.rm = TRUE)
    lower_bound <- quartiles[1] - iqr_multiplier * iqr
    upper_bound <- quartiles[2] + iqr_multiplier * iqr
    
    outlier_mask <- area_values >= lower_bound & area_values <= upper_bound
    area_values <- area_values[outlier_mask]
  }
  
  # Calculate statistics
  n <- length(area_values)
  area_mean <- mean(area_values, na.rm = TRUE)
  area_median <- median(area_values, na.rm = TRUE)
  area_sd <- sd(area_values, na.rm = TRUE)
  area_se <- if (n > 1) area_sd / sqrt(n) else NA_real_
  
  return(list(
    values = area_values,
    mean = area_mean,
    median = area_median,
    std_deviation = area_sd,
    std_error = area_se,
    n = n
  ))
}

#' Calculate ratios between two sets of peak areas
#' 
#' Computes area ratios with optional log transformation and statistics
#' 
#' @param area_set1 list or numeric vector, first set of peak areas
#' @param area_set2 list or numeric vector, second set of peak areas
#' @param use_log logical, whether to use log ratio (default: FALSE)
#' @param calculate_stats logical, whether to compute statistics (default: TRUE)
#' @param remove_outliers logical, whether to remove outliers (default: TRUE)
#' @param iqr_multiplier numeric, IQR multiplier for outlier detection (default: 1.5)
#' @return list with ratios and optional statistics
calculate_area_ratios <- function(area_set1,
                                  area_set2,
                                  use_log = FALSE,
                                  calculate_stats = TRUE,
                                  remove_outliers = TRUE,
                                  iqr_multiplier = 1.5) {
  # Convert to vectors if needed
  if (is.list(area_set1)) area_set1 <- unlist(area_set1)
  if (is.list(area_set2)) area_set2 <- unlist(area_set2)
  
  # Check lengths match
  if (length(area_set1) != length(area_set2)) {
    stop("area_set1 and area_set2 must have the same length")
  }
  
  # Calculate ratios
  if (use_log) {
    ratios <- abs(log(area_set1 / area_set2))
  } else {
    ratios <- abs(area_set1 / area_set2)
  }
  
  # Calculate statistics if requested
  if (calculate_stats) {
    ratio_stats <- calculate_area_statistics(ratios, 
                                            remove_outliers, 
                                            iqr_multiplier)
    names(ratio_stats)[names(ratio_stats) == "values"] <- "ratios"
    return(ratio_stats)
  } else {
    return(list(ratios = ratios))
  }
}

```{r}
# Matched filter approach for peak detection
matched_filter_detection <- function(mz, intensity, 
                                   peak_width = 0.1,  # Expected peak width
                                   threshold = 0.1) {
  
  # Create Gaussian template
  sigma <- peak_width / (2 * sqrt(2 * log(2)))  # Convert FWHM to sigma
  template_size <- ceiling(6 * sigma)  # Template size
  x <- seq(-template_size, template_size, length.out = 2 * template_size + 1)
  template <- exp(-x^2 / (2 * sigma^2))
  template <- template / sum(template)
  
  # Apply matched filter
  filtered <- stats::filter(intensity, template, sides = 2)
  filtered[is.na(filtered)] <- 0
  
  # Threshold and find local maxima
  above_threshold <- filtered > threshold * max(filtered)
  peaks <- logical(length(intensity))
  
  for (i in 2:(length(intensity)-1)) {
    if (above_threshold[i] && 
        filtered[i] > filtered[i-1] && 
        filtered[i] > filtered[i+1]) {
      peaks[i] <- TRUE
    }
  }
  
  return(list(
    peaks = which(peaks),
    filtered_signal = filtered
  ))
}
```

```{r}
# Peak integration function
integrate_peak <- function(mz, intensity, peak_center, integration_method = "trapezoid") {
  # Find peak boundaries
  peak_idx <- which.min(abs(mz - peak_center))
  
  # Simple peak boundary detection
  left_bound <- peak_idx
  right_bound <- peak_idx
  
  # Find left boundary (where intensity drops significantly)
  while (left_bound > 1 && 
         intensity[left_bound - 1] > 0.1 * intensity[peak_idx]) {
    left_bound <- left_bound - 1
  }
  
  # Find right boundary
  while (right_bound < length(intensity) && 
         intensity[right_bound + 1] > 0.1 * intensity[peak_idx]) {
    right_bound <- right_bound + 1
  }
  
  # Integration
  if (integration_method == "trapezoid") {
    # Trapezoidal rule
    mz_region <- mz[left_bound:right_bound]
    int_region <- intensity[left_bound:right_bound]
    
    area <- sum(diff(mz_region) * (int_region[-1] + int_region[-length(int_region)]) / 2)
  } else if (integration_method == "sum") {
    # Simple sum
    area <- sum(intensity[left_bound:right_bound])
  }
  
  return(list(
    area = area,
    peak_height = intensity[peak_idx],
    peak_mz = mz[peak_idx],
    left_bound = left_bound,
    right_bound = right_bound,
    boundaries_mz = c(mz[left_bound], mz[right_bound])
  ))
}
```

```{r}
# Gaussian peak fitting
fit_gaussian_peak <- function(mz, intensity, peak_center) {
  # Extract region around peak
  peak_idx <- which.min(abs(mz - peak_center))
  
  # Define fitting region (Â±20 points around peak)
  start_idx <- max(1, peak_idx - 20)
  end_idx <- min(length(mz), peak_idx + 20)
  
  mz_region <- mz[start_idx:end_idx]
  int_region <- intensity[start_idx:end_idx]
  
  # Gaussian function
  gaussian <- function(x, params) {
    A <- params[1]  # Amplitude
    mu <- params[2] # Mean (center)
    sigma <- params[3] # Standard deviation
    baseline <- params[4] # Baseline
    
    return(baseline + A * exp(-(x - mu)^2 / (2 * sigma^2)))
  }
  
  # Objective function for fitting
  objective <- function(params) {
    predicted <- gaussian(mz_region, params)
    return(sum((int_region - predicted)^2))
  }
  
  # Initial parameters
  initial_params <- c(
    max(int_region),  # Amplitude
    mz_region[which.max(int_region)],  # Center
    0.05,  # Width
    min(int_region)  # Baseline
  )
  
  # Fit (simple optimization)
  tryCatch({
    fit_result <- optim(initial_params, objective, method = "Nelder-Mead")
    fitted_params <- fit_result$par
    
    # Calculate R-squared
    fitted_values <- gaussian(mz_region, fitted_params)
    ss_res <- sum((int_region - fitted_values)^2)
    ss_tot <- sum((int_region - mean(int_region))^2)
    r_squared <- 1 - ss_res / ss_tot
    
    # Calculate area under Gaussian curve
    gaussian_area <- fitted_params[1] * fitted_params[3] * sqrt(2 * pi)
    
    return(list(
      amplitude = fitted_params[1],
      center = fitted_params[2],
      sigma = fitted_params[3],
      baseline = fitted_params[4],
      area = gaussian_area,
      r_squared = r_squared,
      fitted_values = fitted_values,
      mz_region = mz_region
    ))
  }, error = function(e) {
    return(NULL)
  })
}
```

```{r}
assess_peak_symmetry <- function(mz, intensity, peak_center) {
  peak_idx <- which.min(abs(mz - peak_center))
  peak_height <- intensity[peak_idx]
  
  # Find half-maximum points
  half_max <- peak_height / 2
  
  # Find left half-maximum point
  left_idx <- peak_idx
  while (left_idx > 1 && intensity[left_idx] > half_max) {
    left_idx <- left_idx - 1
  }
  
  # Find right half-maximum point
  right_idx <- peak_idx
  while (right_idx < length(intensity) && intensity[right_idx] > half_max) {
    right_idx <- right_idx + 1
  }
  
  # Calculate asymmetry factor
  left_width <- mz[peak_idx] - mz[left_idx]
  right_width <- mz[right_idx] - mz[peak_idx]
  
  asymmetry_factor <- right_width / left_width
  
  return(list(
    asymmetry_factor = asymmetry_factor,
    fwhm = left_width + right_width,
    left_width = left_width,
    right_width = right_width
  ))
}
```

```{r}
#' Filter peaks by minimum distance threshold
#' 
#' Removes closely spaced peaks, keeping only the highest peak in each group
#' 
#' @param peak_data data.frame with peak information (must have 'index' and 'height' columns)
#' @param min_distance numeric, minimum index distance between peaks
#' @param verbose logical, whether to print filtering messages
#' @return filtered peak data.frame
filter_peaks_by_distance <- function(peak_data, min_distance = 8, verbose = FALSE) {
  # Input validation
  if (min_distance < 1) {
    stop("Minimum distance must be greater than 0")
  }
  
  if (!inherits(peak_data, "data.frame") || 
      !all(c("index", "height") %in% names(peak_data))) {
    stop("Input must be a data.frame with 'index' and 'height' columns")
  }
  
  if (nrow(peak_data) == 0) {
    return(peak_data)
  }
  
  # Calculate distances between consecutive peaks
  peak_indices <- peak_data$index
  peak_heights <- peak_data$height
  index_differences <- diff(peak_indices)
  
  # Identify peaks that are too close together
  too_close <- index_differences < min_distance
  
  if (!any(too_close)) {
    if (verbose) {
      message(sprintf("No peaks removed (all peaks separated by >= %d indices)", 
                      min_distance))
    }
    return(peak_data)
  }
  
  # Find groups of adjacent peaks that are too close
  close_indices <- which(too_close)
  peak_groups <- split(close_indices, 
                       cumsum(c(1, diff(close_indices) != 1)))
  
  # For each group, keep only the peak with maximum height
  peaks_to_remove <- integer(0)
  for (group in peak_groups) {
    # Include the peak before the group
    group_start <- if (group[1] > 1) group[1] else 1
    group_end <- group[length(group)] + 1
    
    # Get all peak indices in this group
    group_peak_indices <- group_start:group_end
    
    # Find the peak with maximum height in this group
    group_heights <- peak_heights[group_peak_indices]
    max_height_idx <- group_peak_indices[which.max(group_heights)]
    
    # Mark all others for removal
    peaks_to_remove <- c(peaks_to_remove, 
                         setdiff(group_peak_indices, max_height_idx))
  }
  
  # Remove the selected peaks
  filtered_peaks <- peak_data[-peaks_to_remove, , drop = FALSE]
  num_removed <- length(peaks_to_remove)
  
  if (verbose && num_removed > 0) {
    message(sprintf("%d peak%s removed using distance threshold (d=%d)", 
                    num_removed, 
                    ifelse(num_removed == 1, "", "s"), 
                    min_distance))
  }
  
  return(filtered_peaks)
}

#' Detect peaks in spectral data with configurable parameters
#' 
#' Identifies local maxima in spectral data with neighbor validation,
#' minimum height filtering, and distance-based filtering
#' 
#' @param spectral_data numeric vector of intensity values (can be named with m/z values)
#' @param neighbor_count integer, number of neighbors to check on each side (default: 3)
#' @param min_height numeric, optional minimum peak height threshold
#' @param min_distance numeric, optional minimum index distance between peaks
#' @param peak_type character, label for peak type (default: "peak")
#' @param verbose logical, whether to print processing messages
#' @return data.frame with detected peaks (index, peak position, height, type)
detect_spectral_peaks <- function(spectral_data, 
                                   neighbor_count = 3,
                                   min_height = NULL,
                                   min_distance = NULL,
                                   peak_type = "peak",
                                   verbose = FALSE) {
  # Input validation
  if (!is.numeric(spectral_data) || length(spectral_data) < 3) {
    stop("Spectral data must be a numeric vector with at least 3 points")
  }
  
  if (neighbor_count < 1) {
    stop("Neighbor count must be at least 1")
  }
  
  data_length <- length(spectral_data)
  data_names <- names(spectral_data)
  
  # Step 1: Find all local maxima (simple peak detection)
  # Compare each point with its immediate neighbors
  candidate_indices <- seq.int(2, data_length - 1)
  is_local_maximum <- spectral_data[candidate_indices] > 
                      spectral_data[candidate_indices - 1] &
                      spectral_data[candidate_indices] > 
                      spectral_data[candidate_indices + 1]
  
  peak_candidates <- candidate_indices[is_local_maximum]
  
  # Step 2: Filter by minimum height if specified
  if (!is.null(min_height)) {
    peak_heights <- spectral_data[peak_candidates]
    above_threshold <- peak_heights >= min_height
    peak_candidates <- peak_candidates[above_threshold]
    
    if (verbose) {
      num_filtered <- sum(!above_threshold)
      if (num_filtered > 0) {
        message(sprintf("%d peak%s below minimum height threshold (%.2f)", 
                        num_filtered,
                        ifelse(num_filtered == 1, "", "s"),
                        min_height))
      }
    }
  }
  
  # Step 3: Validate peaks using neighbor checking
  if (neighbor_count > 1 && length(peak_candidates) > 0) {
    # Remove peaks too close to boundaries
    valid_range <- (neighbor_count + 1):(data_length - neighbor_count)
    peak_candidates <- peak_candidates[peak_candidates %in% valid_range]
    
    if (length(peak_candidates) > 0) {
      # Create validation matrix: check n neighbors on each side
      validation_results <- matrix(FALSE, 
                                   nrow = length(peak_candidates), 
                                   ncol = neighbor_count)
      
      for (neighbor_idx in seq_len(neighbor_count)) {
        offset <- neighbor_idx
        # Check that peak is higher than neighbors at this offset
        left_neighbors <- spectral_data[peak_candidates - offset]
        right_neighbors <- spectral_data[peak_candidates + offset]
        peak_values <- spectral_data[peak_candidates]
        
        validation_results[, neighbor_idx] <- 
          peak_values > left_neighbors & peak_values > right_neighbors
      }
      
      # Keep only peaks that pass all neighbor checks
      all_checks_passed <- rowSums(validation_results) == neighbor_count
      validated_peaks <- peak_candidates[all_checks_passed]
    } else {
      validated_peaks <- integer(0)
    }
  } else {
    validated_peaks <- peak_candidates
  }
  
  # Step 4: Create results data frame
  if (length(validated_peaks) > 0) {
    peak_positions <- if (!is.null(data_names)) {
      data_names[validated_peaks]
    } else {
      as.character(validated_peaks)
    }
    
    peak_results <- data.frame(
      index = validated_peaks,
      position = peak_positions,
      height = as.numeric(spectral_data[validated_peaks]),
      type = peak_type,
      stringsAsFactors = FALSE
    )
    
    # Add class attribute for method dispatch
    class(peak_results) <- c("spectral_peaks", "data.frame")
    
    # Step 5: Apply distance filtering if specified
    if (!is.null(min_distance) && min_distance > 0) {
      peak_results <- filter_peaks_by_distance(peak_results, 
                                                min_distance, 
                                                verbose)
    }
    
    return(peak_results)
  } else {
    # Return empty data frame with correct structure
    empty_result <- data.frame(
      index = integer(0),
      position = character(0),
      height = numeric(0),
      type = character(0),
      stringsAsFactors = FALSE
    )
    class(empty_result) <- c("spectral_peaks", "data.frame")
    return(empty_result)
  }
}
```

```{r}
#' Calculate peak slope angles and asymmetry metrics
#' 
#' Analyzes the geometric properties of peaks by calculating
#' slope angles on both sides and measuring peak asymmetry
#' 
#' @param spectral_data numeric vector of intensity values
#' @param peak_data data.frame with peak information (must have 'index' column)
#' @param neighbor_offset integer, distance from peak to measure slope (default: 3)
#' @param use_absolute logical, whether to use absolute angles (default: TRUE)
#' @param decimal_places integer, number of decimal places for rounding (default: 2)
#' @return data.frame with peak data and slope metrics
calculate_peak_slope_metrics <- function(spectral_data,
                                         peak_data,
                                         neighbor_offset = 3,
                                         use_absolute = TRUE,
                                         decimal_places = 2) {
  # Input validation
  if (!inherits(peak_data, "data.frame") || !"index" %in% names(peak_data)) {
    stop("peak_data must be a data.frame with an 'index' column")
  }
  
  if (nrow(peak_data) == 0) {
    return(peak_data)
  }
  
  data_length <- length(spectral_data)
  peak_indices <- peak_data$index
  
  # Initialize result columns
  left_angles <- numeric(nrow(peak_data))
  right_angles <- numeric(nrow(peak_data))
  
  # Calculate slope angles for each peak
  for (i in seq_len(nrow(peak_data))) {
    peak_idx <- peak_indices[i]
    
    # Check if peak is far enough from boundaries
    if (peak_idx <= neighbor_offset || peak_idx > (data_length - neighbor_offset)) {
      left_angles[i] <- NA
      right_angles[i] <- NA
      next
    }
    
    # Get coordinates for slope calculation
    peak_x <- peak_idx
    peak_y <- spectral_data[peak_idx]
    
    # Left side slope
    left_x <- peak_idx - neighbor_offset
    left_y <- spectral_data[left_x]
    left_slope_radians <- atan2(left_y - peak_y, left_x - peak_x)
    left_angles[i] <- left_slope_radians * 180 / pi
    
    # Right side slope
    right_x <- peak_idx + neighbor_offset
    right_y <- spectral_data[right_x]
    right_slope_radians <- atan2(right_y - peak_y, right_x - peak_x)
    right_angles[i] <- right_slope_radians * 180 / pi
  }
  
  # Convert to absolute values if requested
  if (use_absolute) {
    left_angles <- abs(left_angles)
    right_angles <- abs(right_angles)
  }
  
  # Calculate mean slope angle
  mean_angles <- rowMeans(cbind(left_angles, right_angles), na.rm = TRUE)
  
  # Calculate asymmetry (percent difference between left and right slopes)
  # Higher values indicate more asymmetric peaks
  angle_differences <- abs(left_angles - right_angles)
  asymmetry_percent <- ifelse(mean_angles > 0,
                              (angle_differences / mean_angles) * 100,
                              NA)
  
  # Combine results
  result <- cbind(
    peak_data,
    data.frame(
      left_slope_angle = round(left_angles, decimal_places),
      right_slope_angle = round(right_angles, decimal_places),
      mean_slope_angle = round(mean_angles, decimal_places),
      asymmetry_percent = round(asymmetry_percent, decimal_places),
      stringsAsFactors = FALSE
    )
  )
  
  rownames(result) <- NULL
  return(result)
}
```

```{r}
#' 
#' Identifies local minima by inverting the data and applying
#' peak detection algorithms
```

```{r}
#| eval: false

#' Calculate polygon area using the shoelace formula
#' 
#' Computes the area of a polygon defined by x,y coordinates
#' using the shoelace (surveyor's) formula. Useful for calculating
#' peak areas in spectral data.
#' 
#' @param x_coords numeric vector of x-coordinates
#' @param y_coords numeric vector of y-coordinates
#' @return numeric, area of the polygon
calculate_polygon_area <- function(x_coords, y_coords) {
  # Input validation
  if (length(x_coords) != length(y_coords)) {
    stop("x and y coordinate vectors must have the same length")
  }
  
  if (any(is.na(x_coords)) || any(is.na(y_coords))) {
    return(NA_real_)
  }
  
  if (length(x_coords) < 3) {
    return(0)
  }
  
  n_points <- length(x_coords)
  
  # Shoelace formula: area = 0.5 * |sum(x_i * y_{i+1} - x_{i+1} * y_i)|
  # Create circular index for wrapping around
  next_index <- c(seq.int(2, n_points), 1)
  
  # Calculate cross products
  cross_products <- x_coords * y_coords[next_index] - 
                    x_coords[next_index] * y_coords
  
  # Sum and take absolute value, then halve
  polygon_area <- abs(sum(cross_products)) * 0.5
  
  return(polygon_area)
}

#' Refine peak boundaries using linear regression baseline correction
#' 
#' Uses linear regression to identify and remove baseline contributions
#' from peak area calculations, improving accuracy
#' 
#' @param peak_regions list of data.frames, each containing x, y coordinates for a peak region
#' @param rounding_digits integer, number of decimal places for comparison (default: 8)
#' @return list of refined peak region data.frames
refine_peak_boundaries <- function(peak_regions, rounding_digits = 8) {
  if (!is.list(peak_regions) || length(peak_regions) == 0) {
    return(peak_regions)
  }
  
  refined_regions <- vector("list", length(peak_regions))
  
  for (i in seq_along(peak_regions)) {
    region_data <- peak_regions[[i]]
    
    # Skip if not a valid data frame
    if (!inherits(region_data, "data.frame") || nrow(region_data) < 2) {
      refined_regions[[i]] <- region_data
      next
    }
    
    # Extract coordinates
    x_coords <- region_data[[1]]  # First column (x_in or x_p)
    y_coords <- region_data[[3]]  # Third column (y)
    
    # Fit linear regression using first and last points (baseline)
    n_points <- nrow(region_data)
    baseline_points <- data.frame(
      x = c(x_coords[1], x_coords[n_points]),
      y = c(y_coords[1], y_coords[n_points])
    )
    
    baseline_model <- tryCatch(
      lm(y ~ x, data = baseline_points),
      error = function(e) NULL
    )
    
    if (is.null(baseline_model)) {
      refined_regions[[i]] <- region_data
      next
    }
    
    # Predict baseline values for all points
    predicted_baseline <- predict(baseline_model, 
                                   newdata = data.frame(x = x_coords))
    
    # Calculate difference between actual and predicted (baseline-corrected)
    baseline_corrected <- round(y_coords, rounding_digits) - 
                          round(predicted_baseline, rounding_digits)
    
    # Split region into two halves
    midpoint <- ceiling(n_points / 2)
    first_half <- seq_len(midpoint)
    second_half <- seq(midpoint + 1, n_points)
    
    # Find negative values (below baseline) in each half
    negative_first <- which(baseline_corrected[first_half] < 0)
    negative_second <- which(baseline_corrected[second_half] < 0)
    
    # Adjust indices for second half
    if (length(negative_second) > 0) {
      negative_second <- negative_second + midpoint
    }
    
    # Remove points below baseline, keeping the peak region
    indices_to_keep <- seq_len(n_points)
    
    # Remove from end (second half) first
    if (length(negative_second) > 0) {
      first_negative_end <- min(negative_second)
      indices_to_keep <- indices_to_keep[indices_to_keep < first_negative_end]
    }
    
    # Remove from beginning (first half)
    if (length(negative_first) > 0) {
      last_negative_start <- max(negative_first)
      indices_to_keep <- indices_to_keep[indices_to_keep > last_negative_start]
    }
    
    # Create refined region
    if (length(indices_to_keep) >= 2) {
      refined_regions[[i]] <- region_data[indices_to_keep, , drop = FALSE]
    } else {
      refined_regions[[i]] <- region_data
    }
  }
  
  return(refined_regions)
}

#' Calculate peak areas for specified peaks in spectral data matrix
#' 
#' Detects peaks and troughs, then calculates areas for user-specified peaks
#' using the region between adjacent troughs
#' 
#' @param spectral_matrix matrix or data.frame, columns are spectra, rows are m/z values
#' @param target_peak character or numeric, peak identifier to calculate area for
#' @param neighbor_count integer, number of neighbors for peak detection (default: 3)
#' @param coordinate_type character, use "index" or "position" for area calculation (default: "index")
#' @param refine_baseline logical, whether to refine boundaries using regression (default: TRUE)
#' @param rounding_precision integer, decimal places for baseline refinement (default: 8)
#' @param ... additional arguments passed to peak detection function
#' @return list with peak coordinates, calculated areas, and detection results
calculate_peak_areas <- function(spectral_matrix,
                                 target_peak,
                                 neighbor_count = 3,
                                 coordinate_type = "index",
                                 refine_baseline = TRUE,
                                 rounding_precision = 8,
                                 ...) {
  # Input validation
  if (!is.matrix(spectral_matrix) && !is.data.frame(spectral_matrix)) {
    stop("spectral_matrix must be a matrix or data.frame")
  }
  
  # Calculate spectral resolution (m/z step size)
  row_names <- rownames(spectral_matrix)
  if (is.null(row_names) || length(row_names) < 2) {
    stop("spectral_matrix must have row names representing m/z values")
  }
  
  spectral_resolution <- abs(as.numeric(row_names[1]) - as.numeric(row_names[2]))
  n_spectra <- ncol(spectral_matrix)
  
  # Detect peaks and troughs for each spectrum
  detected_peaks <- apply(spectral_matrix, 2, function(spectrum) {
    detect_spectral_peaks(spectrum, neighbor_count = neighbor_count, 
                         verbose = FALSE, ...)
  }, simplify = FALSE)
  
  detected_troughs <- apply(spectral_matrix, 2, function(spectrum) {
    detect_spectral_troughs(spectrum, neighbor_count = 1, 
                           rename_column = FALSE)
  }, simplify = FALSE)
  
  # Combine peaks and troughs, sort by index
  combined_features <- mapply(function(peaks, troughs) {
    combined <- rbind(peaks, troughs)
    combined <- combined[order(combined$index), ]
    rownames(combined) <- NULL
    return(combined)
  }, detected_peaks, detected_troughs, SIMPLIFY = FALSE)
  
  # Find target peak in each spectrum
  peak_regions <- vector("list", n_spectra)
  calculated_areas <- vector("list", n_spectra)
  
  for (spec_idx in seq_len(n_spectra)) {
    features <- combined_features[[spec_idx]]
    
    # Find target peak
    peak_matches <- grep(as.character(target_peak), features$position, 
                        value = FALSE, fixed = TRUE)
    peak_rows <- features[peak_matches, , drop = FALSE]
    peak_rows <- peak_rows[peak_rows$type == "peak", , drop = FALSE]
    
    if (nrow(peak_rows) == 0) {
      peak_regions[[spec_idx]] <- NULL
      calculated_areas[[spec_idx]] <- NA_real_
      next
    }
    
    # Use first matching peak
    target_peak_idx <- peak_rows$index[1]
    feature_indices <- features$index
    
    # Find adjacent troughs
    peak_position <- which(feature_indices == target_peak_idx)
    left_trough_idx <- if (peak_position > 1) {
      features$index[peak_position - 1]
    } else {
      1
    }
    right_trough_idx <- if (peak_position < length(feature_indices)) {
      features$index[peak_position + 1]
    } else {
      nrow(spectral_matrix)
    }
    
    # Extract region coordinates
    region_indices <- left_trough_idx:right_trough_idx
    region_mz <- as.numeric(row_names[region_indices])
    region_intensity <- spectral_matrix[region_indices, spec_idx]
    
    # Create region data frame
    region_data <- data.frame(
      index = region_indices,
      mz = region_mz,
      intensity = region_intensity,
      stringsAsFactors = FALSE
    )
    
    peak_regions[[spec_idx]] <- region_data
    
    # Refine boundaries if requested
    if (refine_baseline) {
      refined_list <- refine_peak_boundaries(list(region_data), 
                                             rounding_precision)
      region_data <- refined_list[[1]]
    }
    
    # Calculate area using appropriate coordinates
    if (coordinate_type == "index") {
      x_coords <- region_data$index
    } else {
      x_coords <- region_data$mz
    }
    y_coords <- region_data$intensity
    
    calculated_areas[[spec_idx]] <- calculate_polygon_area(x_coords, y_coords)
  }
  
  return(list(
    peak_regions = peak_regions,
    areas = calculated_areas,
    detected_peaks = detected_peaks,
    detected_troughs = detected_troughs
  ))
}

#' Calculate peak area between two specified m/z values
#' 
#' Manually defines integration region and calculates area
#' 
#' @param spectral_data numeric vector, intensity values (named with m/z)
#' @param mz_range numeric vector of length 2, [lower, upper] m/z bounds
#' @param coordinate_type character, "index" or "position" for area calculation
#' @param refine_baseline logical, whether to refine boundaries (default: TRUE)
#' @param rounding_precision integer, decimal places for refinement (default: 8)
#' @return list with region coordinates and calculated area
calculate_area_between_bands <- function(spectral_data,
                                         mz_range,
                                         coordinate_type = "index",
                                         refine_baseline = TRUE,
                                         rounding_precision = 8) {
  # Input validation
  if (length(mz_range) != 2) {
    stop("mz_range must be a vector of length 2: [lower, upper]")
  }
  
  if (mz_range[1] >= mz_range[2]) {
    stop("mz_range[1] must be less than mz_range[2]")
  }
  
  mz_names <- names(spectral_data)
  if (is.null(mz_names)) {
    stop("spectral_data must be named with m/z values")
  }
  
  # Find indices for m/z range
  mz_values <- as.numeric(mz_names)
  lower_idx <- which.min(abs(mz_values - mz_range[1]))
  upper_idx <- which.min(abs(mz_values - mz_range[2]))
  
  # Extract region
  region_indices <- lower_idx:upper_idx
  region_mz <- mz_values[region_indices]
  region_intensity <- spectral_data[region_indices]
  
  # Create region data frame
  region_data <- data.frame(
    index = region_indices,
    mz = region_mz,
    intensity = region_intensity,
    stringsAsFactors = FALSE
  )
  
  # Refine if requested
  if (refine_baseline) {
    refined_list <- refine_peak_boundaries(list(region_data), 
                                          rounding_precision)
    region_data <- refined_list[[1]]
  }
  
  # Calculate area
  if (coordinate_type == "index") {
    x_coords <- region_data$index
  } else {
    x_coords <- region_data$mz
  }
  y_coords <- region_data$intensity
  
  calculated_area <- calculate_polygon_area(x_coords, y_coords)
  
  return(list(
    region_coordinates = region_data,
    area = calculated_area
  ))
}

#' Calculate descriptive statistics for peak areas
#' 
#' Computes mean, median, standard deviation, and standard error
#' with optional outlier removal
#' 
#' @param area_values numeric vector or list, peak area values
#' @param remove_outliers logical, whether to remove outliers (default: TRUE)
#' @param iqr_multiplier numeric, IQR multiplier for outlier detection (default: 1.5)
#' @return list with area values and statistics
calculate_area_statistics <- function(area_values,
                                      remove_outliers = TRUE,
                                      iqr_multiplier = 1.5) {
  # Convert to vector if needed
  if (is.list(area_values)) {
    area_values <- unlist(area_values)
  }
  
  # Remove infinite values
  area_values <- area_values[is.finite(area_values)]
  
  if (length(area_values) == 0) {
    return(list(
      values = numeric(0),
      mean = NA_real_,
      median = NA_real_,
      std_deviation = NA_real_,
      std_error = NA_real_
    ))
  }
  
  # Remove outliers using IQR method
  if (remove_outliers && length(area_values) > 4) {
    quartiles <- quantile(area_values, probs = c(0.25, 0.75), na.rm = TRUE)
    iqr <- IQR(area_values, na.rm = TRUE)
    lower_bound <- quartiles[1] - iqr_multiplier * iqr
    upper_bound <- quartiles[2] + iqr_multiplier * iqr
    
    outlier_mask <- area_values >= lower_bound & area_values <= upper_bound
    area_values <- area_values[outlier_mask]
  }
  
  # Calculate statistics
  n <- length(area_values)
  area_mean <- mean(area_values, na.rm = TRUE)
  area_median <- median(area_values, na.rm = TRUE)
  area_sd <- sd(area_values, na.rm = TRUE)
  area_se <- if (n > 1) area_sd / sqrt(n) else NA_real_
  
  return(list(
    values = area_values,
    mean = area_mean,
    median = area_median,
    std_deviation = area_sd,
    std_error = area_se,
    n = n
  ))
}

#' Calculate ratios between two sets of peak areas
#' 
#' Computes area ratios with optional log transformation and statistics
#' 
#' @param area_set1 list or numeric vector, first set of peak areas
#' @param area_set2 list or numeric vector, second set of peak areas
#' @param use_log logical, whether to use log ratio (default: FALSE)
#' @param calculate_stats logical, whether to compute statistics (default: TRUE)
#' @param remove_outliers logical, whether to remove outliers (default: TRUE)
#' @param iqr_multiplier numeric, IQR multiplier for outlier detection (default: 1.5)
#' @return list with ratios and optional statistics
calculate_area_ratios <- function(area_set1,
                                  area_set2,
                                  use_log = FALSE,
                                  calculate_stats = TRUE,
                                  remove_outliers = TRUE,
                                  iqr_multiplier = 1.5) {
  # Convert to vectors if needed
  if (is.list(area_set1)) area_set1 <- unlist(area_set1)
  if (is.list(area_set2)) area_set2 <- unlist(area_set2)
  
  # Check lengths match
  if (length(area_set1) != length(area_set2)) {
    stop("area_set1 and area_set2 must have the same length")
  }
  
  # Calculate ratios
  if (use_log) {
    ratios <- abs(log(area_set1 / area_set2))
  } else {
    ratios <- abs(area_set1 / area_set2)
  }
  
  # Calculate statistics if requested
  if (calculate_stats) {
    ratio_stats <- calculate_area_statistics(ratios, 
                                            remove_outliers, 
                                            iqr_multiplier)
    names(ratio_stats)[names(ratio_stats) == "values"] <- "ratios"
    return(ratio_stats)
  } else {
    return(list(ratios = ratios))
  }
}
```
