1  R Fundamentals for Mass Spectrometry

This chapter introduces R programming concepts specifically relevant to mass spectrometry data analysis.

1.1 R Environment Setup

1.1.1 Installing Required Packages

The R for Mass Spectrometry ecosystem is built on Bioconductor, a collection of R packages for biological data analysis.

# Step 1: Install BiocManager if not already installed
if (!require("BiocManager", quietly = TRUE))
    install.packages("BiocManager")

# Step 2: Install core R for Mass Spectrometry packages
BiocManager::install(c(
  # Core MS data infrastructure
  "Spectra",              # Modern MS data handling
  "MsBackendMzR",        # Reading raw data files
  "MsBackendSql",        # SQL-based backend
  "MsBackendHdf5Peaks",  # HDF5 storage backend
  
  # MS data utilities
  "msdata",              # Example MS datasets
  "MsDataHub",           # Access to online MS resources
  "ProtGenerics",        # Generic functions for proteomics
  "MsCoreUtils",         # Core MS utilities
  
  # Proteomics packages
  "PSMatch",             # Peptide-spectrum matching
  "QFeatures",           # Quantitative features
  
  # Metabolomics packages
  "xcms",                # LC/MS data processing
  "CAMERA",              # Annotation of adducts and isotopes
  "MetaboCoreUtils",     # Metabolomics utilities
  "CompoundDb",          # Compound databases
  
  # Statistical analysis
  "limma",               # Linear models for microarray/proteomics
  "DEqMS"                # Differential expression
))

# Step 3: Install CRAN packages for data manipulation and visualization
install.packages(c(
  "tidyverse",           # Data science ecosystem
  "ggplot2",             # Advanced plotting
  "plotly",              # Interactive plots
  "pheatmap",            # Heatmaps
  "corrplot",            # Correlation plots
  "BiocParallel"         # Parallel processing
))

# Check if key packages are installed correctly
required_packages <- c("Spectra", "QFeatures", "xcms", "tidyverse")

}

1.2 Understanding R for Mass Spectrometry Ecosystem

1.2.1 Package Architecture

The R for Mass Spectrometry initiative provides a modular ecosystem:

  • Core Infrastructure: Spectra, QFeatures for data structures
  • Data Access: MsBackendMzR, MsBackendSql for reading files
  • Proteomics: PSMatch, ProtGenerics for peptide/protein analysis
  • Metabolomics: xcms, CAMERA for small molecule analysis
  • Utilities: MsCoreUtils, MetaboCoreUtils for common operations
# Display available backends for Spectra
available_backends <- c(
  "MsBackendMzR" = "Read mzML/mzXML files",
  "MsBackendDataFrame" = "In-memory storage",
  "MsBackendHdf5Peaks" = "HDF5-based storage",
  "MsBackendSql" = "SQL database storage"
)

cat("Available Spectra Backends:\n")
Available Spectra Backends:
for (backend in names(available_backends)) {
  cat("  •", backend, "-", available_backends[backend], "\n")
}
  • MsBackendMzR - Read mzML/mzXML files 
  • MsBackendDataFrame - In-memory storage 
  • MsBackendHdf5Peaks - HDF5-based storage 
  • MsBackendSql - SQL database storage

1.3 Data Structures in R for MS

1.3.1 Vectors and Matrices

Mass spectra are fundamentally collections of m/z and intensity pairs, which map naturally to R’s vector and matrix structures.

# Example: Creating a simple spectrum
mz_values <- c(100.1, 200.2, 300.3, 400.4)
intensity_values <- c(1000, 2500, 800, 1200)

# Create a simple spectrum data frame
spectrum_df <- data.frame(
  mz = mz_values,
  intensity = intensity_values
)

print(spectrum_df)
     mz intensity
1 100.1      1000
2 200.2      2500
3 300.3       800
4 400.4      1200

# Example: Reading mzML files using Spectra
library(msdata)

# Get path to example file
ms_file <- msdata::proteomics(full.names = TRUE)[1]
cat("Example file:", basename(ms_file), "\n")
Example file: MRM-standmix-5.mzML.gz 
# Load the data with error handling for mzR compatibility issues
tryCatch({
  ms_data <- Spectra(ms_file, backend = MsBackendMzR())
  # Convert to in-memory for better compatibility
  ms_data <- setBackend(ms_data, backend = MsBackendDataFrame())
  cat("\nSuccessfully loaded real MS data\n")
}, error = function(e) {
  cat("\nNote: mzR compatibility issue detected, using synthetic data\n")
  cat("Error:", conditionMessage(e), "\n\n")
  
  # Create synthetic data as fallback
  set.seed(42)
  n_spectra <- 50
  
  library(S4Vectors)
  library(IRanges)
  
  mz_list <- lapply(1:n_spectra, function(i) {
    sort(runif(sample(50:150, 1), 100, 2000))
  })
  
  intensity_list <- lapply(mz_list, function(mz_vals) {
    rlnorm(length(mz_vals), meanlog = 8, sdlog = 2)
  })
  
  # Generate MS levels first
  ms_levels <- sample(1:2, n_spectra, replace = TRUE, prob = c(0.3, 0.7))
  
  spd <- DataFrame(
    msLevel = ms_levels,
    rtime = seq(100, 3000, length.out = n_spectra),
    precursorMz = ifelse(ms_levels == 1, NA_real_, runif(n_spectra, 400, 1500)),
    polarity = rep(1L, n_spectra)
  )
  
  spd$mz <- NumericList(mz_list)
  spd$intensity <- NumericList(intensity_list)
  
  backend <- MsBackendDataFrame()
  backend <- backendInitialize(backend, spd)
  ms_data <<- Spectra(backend)
})

1.4.3 Exporting Data

# Export spectra to different formats

# Export to mzML
export(ms_data[1:10], file = "subset.mzML")

# Export to MGF (for MS2 spectra)
ms2_data <- filterMsLevel(ms_data, 2)
export(ms2_data, file = "ms2_spectra.mgf")

# Export metadata to CSV
metadata <- spectraData(ms_data) %>%
  as.data.frame() %>%
  select(msLevel, rtime, precursorMz, precursorCharge)

write.csv(metadata, "spectra_metadata.csv", row.names = FALSE)
