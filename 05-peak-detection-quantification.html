# Using a simple implementation since MsCoreUtils::noise has specific requirements
estimate_noise_mad <- function(x) {
  # Use MAD (Median Absolute Deviation) for robust noise estimation
  median_val <- median(x, na.rm = TRUE)
  mad_val <- median(abs(x - median_val), na.rm = TRUE)
  # MAD-based noise estimate
  return(mad_val * 1.4826)  # Scale factor for normal distribution
}

# Matched filter approach for peak detection
matched_filter_detection <- function(mz, intensity, 
                                   peak_width = 0.1,  # Expected peak width
                                   threshold = 0.1) {
  
  # Create Gaussian template
  sigma <- peak_width / (2 * sqrt(2 * log(2)))  # Convert FWHM to sigma
  template_size <- ceiling(6 * sigma)  # Template size
  x <- seq(-template_size, template_size, length.out = 2 * template_size + 1)
  template <- exp(-x^2 / (2 * sigma^2))
  template <- template / sum(template)
  
  # Apply matched filter
  filtered <- stats::filter(intensity, template, sides = 2)
  filtered[is.na(filtered)] <- 0
  
  # Threshold and find local maxima
  above_threshold <- filtered > threshold * max(filtered)
  peaks <- logical(length(intensity))
  
  for (i in 2:(length(intensity)-1)) {
    if (above_threshold[i] && 
        filtered[i] > filtered[i-1] && 
        filtered[i] > filtered[i+1]) {
      peaks[i] <- TRUE
    }
  }
  
  return(list(
    peaks = which(peaks),
    filtered_signal = filtered
  ))
}

# Peak integration function
integrate_peak <- function(mz, intensity, peak_center, integration_method = "trapezoid") {
  # Find peak boundaries
  peak_idx <- which.min(abs(mz - peak_center))
  
  # Simple peak boundary detection
  left_bound <- peak_idx
  right_bound <- peak_idx
  
  # Find left boundary (where intensity drops significantly)
  while (left_bound > 1 && 
         intensity[left_bound - 1] > 0.1 * intensity[peak_idx]) {
    left_bound <- left_bound - 1
  }
  
  # Find right boundary
  while (right_bound < length(intensity) && 
         intensity[right_bound + 1] > 0.1 * intensity[peak_idx]) {
    right_bound <- right_bound + 1
  }
  
  # Integration
  if (integration_method == "trapezoid") {
    # Trapezoidal rule
    mz_region <- mz[left_bound:right_bound]
    int_region <- intensity[left_bound:right_bound]
    
    area <- sum(diff(mz_region) * (int_region[-1] + int_region[-length(int_region)]) / 2)
  } else if (integration_method == "sum") {
    # Simple sum
    area <- sum(intensity[left_bound:right_bound])
  }
  
  return(list(
    area = area,
    peak_height = intensity[peak_idx],
    peak_mz = mz[peak_idx],
    left_bound = left_bound,
    right_bound = right_bound,
    boundaries_mz = c(mz[left_bound], mz[right_bound])
  ))
}

# Gaussian peak fitting
fit_gaussian_peak <- function(mz, intensity, peak_center) {
  # Extract region around peak
  peak_idx <- which.min(abs(mz - peak_center))
  
  # Define fitting region (Â±20 points around peak)
  start_idx <- max(1, peak_idx - 20)
  end_idx <- min(length(mz), peak_idx + 20)
  
  mz_region <- mz[start_idx:end_idx]
  int_region <- intensity[start_idx:end_idx]
  
  # Gaussian function
  gaussian <- function(x, params) {
    A <- params[1]  # Amplitude
    mu <- params[2] # Mean (center)
    sigma <- params[3] # Standard deviation
    baseline <- params[4] # Baseline
    
    return(baseline + A * exp(-(x - mu)^2 / (2 * sigma^2)))
  }
  
  # Objective function for fitting
  objective <- function(params) {
    predicted <- gaussian(mz_region, params)
    return(sum((int_region - predicted)^2))
  }
  
  # Initial parameters
  initial_params <- c(
    max(int_region),  # Amplitude
    mz_region[which.max(int_region)],  # Center
    0.05,  # Width
    min(int_region)  # Baseline
  )
  
  # Fit (simple optimization)
  tryCatch({
    fit_result <- optim(initial_params, objective, method = "Nelder-Mead")
    fitted_params <- fit_result$par
    
    # Calculate R-squared
    fitted_values <- gaussian(mz_region, fitted_params)
    ss_res <- sum((int_region - fitted_values)^2)
    ss_tot <- sum((int_region - mean(int_region))^2)
    r_squared <- 1 - ss_res / ss_tot
    
    # Calculate area under Gaussian curve
    gaussian_area <- fitted_params[1] * fitted_params[3] * sqrt(2 * pi)
    
    return(list(
      amplitude = fitted_params[1],
      center = fitted_params[2],
      sigma = fitted_params[3],
      baseline = fitted_params[4],
      area = gaussian_area,
      r_squared = r_squared,
      fitted_values = fitted_values,
      mz_region = mz_region
    ))
  }, error = function(e) {
    return(NULL)
  })
}

calculate_snr <- function(mz, intensity, peak_center, noise_region_width = 2.0) {
  peak_idx <- which.min(abs(mz - peak_center))
  peak_intensity <- intensity[peak_idx]
  
  # Define noise region (away from peak)
  noise_indices <- which(abs(mz - peak_center) > noise_region_width)
  
  if (length(noise_indices) > 10) {
    noise_level <- sd(intensity[noise_indices])
    snr <- peak_intensity / noise_level
  } else {
    # Fallback: use entire spectrum for noise estimation
    snr <- peak_intensity / sd(intensity)
  }
  
  return(snr)
}

assess_peak_symmetry <- function(mz, intensity, peak_center) {
  peak_idx <- which.min(abs(mz - peak_center))
  peak_height <- intensity[peak_idx]
  
  # Find half-maximum points
  half_max <- peak_height / 2
  
  # Find left half-maximum point
  left_idx <- peak_idx
  while (left_idx > 1 && intensity[left_idx] > half_max) {
    left_idx <- left_idx - 1
  }
  
  # Find right half-maximum point
  right_idx <- peak_idx
  while (right_idx < length(intensity) && intensity[right_idx] > half_max) {
    right_idx <- right_idx + 1
  }
  
  # Calculate asymmetry factor
  left_width <- mz[peak_idx] - mz[left_idx]
  right_width <- mz[right_idx] - mz[peak_idx]
  
  asymmetry_factor <- right_width / left_width
  
  return(list(
    asymmetry_factor = asymmetry_factor,
    fwhm = left_width + right_width,
    left_width = left_width,
    right_width = right_width
  ))
}

create_peak_list <- function(mz, intensity, peak_indices) {
  peak_data <- data.frame(
    peak_id = seq_along(peak_indices),
    mz = mz[peak_indices],
    intensity = intensity[peak_indices],
    relative_intensity = intensity[peak_indices] / max(intensity) * 100
  )
  
  # Add quality metrics
  peak_data$snr <- sapply(peak_indices, function(idx) {
    calculate_snr(mz, intensity, mz[idx])
  })
  
  peak_data$asymmetry <- sapply(peak_indices, function(idx) {
    sym_result <- assess_peak_symmetry(mz, intensity, mz[idx])
    return(sym_result$asymmetry_factor)
  })
  
  # Add area calculations
  peak_data$area <- sapply(peak_indices, function(idx) {
    integration_result <- integrate_peak(mz, intensity, mz[idx])
    return(integration_result$area)
  })
  
  return(peak_data)
}

process_multiple_spectra <- function(spectra_obj, max_spectra = 10) {
  all_peak_lists <- list()
  
  for (i in 1:min(length(spectra_obj), max_spectra)) {
    # Extract spectrum data
    mz_vals <- mz(spectra_obj[i])[[1]]
    int_vals <- intensity(spectra_obj[i])[[1]]
    
    # Skip empty spectra
    if (length(mz_vals) == 0 || max(int_vals) == 0) next
    
    # Detect peaks
    mf_result <- matched_filter_detection(mz_vals, int_vals)
    
    if (length(mf_result$peaks) > 0) {
      # Create peak list
      peak_list <- create_peak_list(mz_vals, int_vals, mf_result$peaks)
      peak_list$spectrum_id <- i
      peak_list$retention_time <- rtime(spectra_obj[i])
      
      all_peak_lists[[i]] <- peak_list
    }
  }
  
  # Combine all peak lists
  combined_peaks <- do.call(rbind, all_peak_lists)
  return(combined_peaks)
}

estimate_noise_mad <- function(x) {
  # Use MAD (Median Absolute Deviation) for robust noise estimation
  median_val <- median(x, na.rm = TRUE)
  mad_val <- median(abs(x - median_val), na.rm = TRUE)
  # MAD-based noise estimate
  return(mad_val * 1.4826)  # Scale factor for normal distribution
}

# Matched filter approach for peak detection
matched_filter_detection <- function(mz, intensity, 
                                   peak_width = 0.1,  # Expected peak width
                                   threshold = 0.1) {
  
  # Create Gaussian template
  sigma <- peak_width / (2 * sqrt(2 * log(2)))  # Convert FWHM to sigma
  template_size <- ceiling(6 * sigma)  # Template size
  x <- seq(-template_size, template_size, length.out = 2 * template_size + 1)
  template <- exp(-x^2 / (2 * sigma^2))
  template <- template / sum(template)
  
  # Apply matched filter
  filtered <- stats::filter(intensity, template, sides = 2)
  filtered[is.na(filtered)] <- 0
  
  # Threshold and find local maxima
  above_threshold <- filtered > threshold * max(filtered)
  peaks <- logical(length(intensity))
  
  for (i in 2:(length(intensity)-1)) {
    if (above_threshold[i] && 
        filtered[i] > filtered[i-1] && 
        filtered[i] > filtered[i+1]) {
      peaks[i] <- TRUE
    }
  }
  
  return(list(
    peaks = which(peaks),
    filtered_signal = filtered
  ))
}

# Peak integration function
integrate_peak <- function(mz, intensity, peak_center, integration_method = "trapezoid") {
  # Find peak boundaries
  peak_idx <- which.min(abs(mz - peak_center))
  
  # Simple peak boundary detection
  left_bound <- peak_idx
  right_bound <- peak_idx
  
  # Find left boundary (where intensity drops significantly)
  while (left_bound > 1 && 
         intensity[left_bound - 1] > 0.1 * intensity[peak_idx]) {
    left_bound <- left_bound - 1
  }
  
  # Find right boundary
  while (right_bound < length(intensity) && 
         intensity[right_bound + 1] > 0.1 * intensity[peak_idx]) {
    right_bound <- right_bound + 1
  }
  
  # Integration
  if (integration_method == "trapezoid") {
    # Trapezoidal rule
    mz_region <- mz[left_bound:right_bound]
    int_region <- intensity[left_bound:right_bound]
    
    area <- sum(diff(mz_region) * (int_region[-1] + int_region[-length(int_region)]) / 2)
  } else if (integration_method == "sum") {
    # Simple sum
    area <- sum(intensity[left_bound:right_bound])
  }
  
  return(list(
    area = area,
    peak_height = intensity[peak_idx],
    peak_mz = mz[peak_idx],
    left_bound = left_bound,
    right_bound = right_bound,
    boundaries_mz = c(mz[left_bound], mz[right_bound])
  ))
}

# Gaussian peak fitting
fit_gaussian_peak <- function(mz, intensity, peak_center) {
  # Extract region around peak
  peak_idx <- which.min(abs(mz - peak_center))
  
  # Define fitting region (Â±20 points around peak)
  start_idx <- max(1, peak_idx - 20)
  end_idx <- min(length(mz), peak_idx + 20)
  
  mz_region <- mz[start_idx:end_idx]
  int_region <- intensity[start_idx:end_idx]
  
  # Gaussian function
  gaussian <- function(x, params) {
    A <- params[1]  # Amplitude
    mu <- params[2] # Mean (center)
    sigma <- params[3] # Standard deviation
    baseline <- params[4] # Baseline
    
    return(baseline + A * exp(-(x - mu)^2 / (2 * sigma^2)))
  }
  
  # Objective function for fitting
  objective <- function(params) {
    predicted <- gaussian(mz_region, params)
    return(sum((int_region - predicted)^2))
  }
  
  # Initial parameters
  initial_params <- c(
    max(int_region),  # Amplitude
    mz_region[which.max(int_region)],  # Center
    0.05,  # Width
    min(int_region)  # Baseline
  )
  
  # Fit (simple optimization)
  tryCatch({
    fit_result <- optim(initial_params, objective, method = "Nelder-Mead")
    fitted_params <- fit_result$par
    
    # Calculate R-squared
    fitted_values <- gaussian(mz_region, fitted_params)
    ss_res <- sum((int_region - fitted_values)^2)
    ss_tot <- sum((int_region - mean(int_region))^2)
    r_squared <- 1 - ss_res / ss_tot
    
    # Calculate area under Gaussian curve
    gaussian_area <- fitted_params[1] * fitted_params[3] * sqrt(2 * pi)
    
    return(list(
      amplitude = fitted_params[1],
      center = fitted_params[2],
      sigma = fitted_params[3],
      baseline = fitted_params[4],
      area = gaussian_area,
      r_squared = r_squared,
      fitted_values = fitted_values,
      mz_region = mz_region
    ))
  }, error = function(e) {
    return(NULL)
  })
}

calculate_snr <- function(mz, intensity, peak_center, noise_region_width = 2.0) {
  peak_idx <- which.min(abs(mz - peak_center))
  peak_intensity <- intensity[peak_idx]
  
  # Define noise region (away from peak)
  noise_indices <- which(abs(mz - peak_center) > noise_region_width)
  
  if (length(noise_indices) > 10) {
    noise_level <- sd(intensity[noise_indices])
    snr <- peak_intensity / noise_level
  } else {
    # Fallback: use entire spectrum for noise estimation
    snr <- peak_intensity / sd(intensity)
  }
  
  return(snr)
}

assess_peak_symmetry <- function(mz, intensity, peak_center) {
  peak_idx <- which.min(abs(mz - peak_center))
  peak_height <- intensity[peak_idx]
  
  # Find half-maximum points
  half_max <- peak_height / 2
  
  # Find left half-maximum point
  left_idx <- peak_idx
  while (left_idx > 1 && intensity[left_idx] > half_max) {
    left_idx <- left_idx - 1
  }
  
  # Find right half-maximum point
  right_idx <- peak_idx
  while (right_idx < length(intensity) && intensity[right_idx] > half_max) {
    right_idx <- right_idx + 1
  }
  
  # Calculate asymmetry factor
  left_width <- mz[peak_idx] - mz[left_idx]
  right_width <- mz[right_idx] - mz[peak_idx]
  
  asymmetry_factor <- right_width / left_width
  
  return(list(
    asymmetry_factor = asymmetry_factor,
    fwhm = left_width + right_width,
    left_width = left_width,
    right_width = right_width
  ))
}

# Assess symmetry for first peak
if (length(mf_result$peaks) > 0) {

create_peak_list <- function(mz, intensity, peak_indices) {
  peak_data <- data.frame(
    peak_id = seq_along(peak_indices),
    mz = mz[peak_indices],
    intensity = intensity[peak_indices],
    relative_intensity = intensity[peak_indices] / max(intensity) * 100
  )
  
  # Add quality metrics
  peak_data$snr <- sapply(peak_indices, function(idx) {
    calculate_snr(mz, intensity, mz[idx])
  })
  
  peak_data$asymmetry <- sapply(peak_indices, function(idx) {
    sym_result <- assess_peak_symmetry(mz, intensity, mz[idx])
    return(sym_result$asymmetry_factor)
  })
  
  # Add area calculations
  peak_data$area <- sapply(peak_indices, function(idx) {
    integration_result <- integrate_peak(mz, intensity, mz[idx])
    return(integration_result$area)
  })
  
  return(peak_data)

process_multiple_spectra <- function(spectra_obj, max_spectra = 10) {
  all_peak_lists <- list()
  
  for (i in 1:min(length(spectra_obj), max_spectra)) {
    # Extract spectrum data
    mz_vals <- mz(spectra_obj[i])[[1]]
    int_vals <- intensity(spectra_obj[i])[[1]]
    
    # Skip empty spectra
    if (length(mz_vals) == 0 || max(int_vals) == 0) next
    
    # Detect peaks
    mf_result <- matched_filter_detection(mz_vals, int_vals)
    
    if (length(mf_result$peaks) > 0) {
      # Create peak list
      peak_list <- create_peak_list(mz_vals, int_vals, mf_result$peaks)
      peak_list$spectrum_id <- i
      peak_list$retention_time <- rtime(spectra_obj[i])
      
      all_peak_lists[[i]] <- peak_list
    }
  }
  
  # Combine all peak lists
  combined_peaks <- do.call(rbind, all_peak_lists)
  return(combined_peaks)
}
