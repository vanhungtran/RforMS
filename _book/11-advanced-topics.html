# Demonstrate memory efficient workflows
large_dataset_workflow <- function(sps_data) {
  # Process spectra in batches without loading all into memory at once
  batch_size <- 20
  n_spectra <- length(sps_data)
  n_batches <- ceiling(n_spectra / batch_size)
  
  results <- list()
  
  for (i in 1:n_batches) {
    start_idx <- (i - 1) * batch_size + 1
    end_idx <- min(i * batch_size, n_spectra)
    
    # Process batch
    batch_sps <- sps_data[start_idx:end_idx]
    
    # Extract summary statistics
    results[[i]] <- data.frame(
      batch = i,
      n_spectra = length(batch_sps),
      rt_range = paste(round(range(rtime(batch_sps)), 2), collapse = "-"),
      ms_levels = paste(unique(msLevel(batch_sps)), collapse = ","),
      mean_peaks = round(mean(lengths(mz(batch_sps))))
    )
    
    # Explicitly remove batch objects
    rm(batch_sps)
    gc()  # Force garbage collection
  }
  
  do.call(rbind, results)
}

custom_spectral_processing <- function(sps) {
  # Custom processing pipeline
  processed_sps <- sps
  
  # 1. Noise reduction with Savitzky-Golay smoothing
  processed_sps <- smooth(processed_sps, method = "SavitzkyGolay")
  
  # 2. Peak picking with custom parameters
  processed_sps <- pickPeaks(processed_sps, 
                            method = "MAD",
                            snr = 3)
  
  # 3. Normalize intensities by dividing by max intensity
  processed_sps <- addProcessing(processed_sps, function(x) {
    x[, "intensity"] <- x[, "intensity"] / max(x[, "intensity"], na.rm = TRUE)
    x
  })
  
  return(processed_sps)
}

calculate_spectral_similarity_matrix <- function(sps, method = "cosine") {
  n <- length(sps)
  similarity_matrix <- matrix(0, nrow = n, ncol = n)
  
  for (i in 1:n) {
    for (j in i:n) {
      if (i == j) {
        similarity_matrix[i, j] <- 1.0
      } else {
        # Calculate pairwise similarity
        sim <- compareSpectra(sps[i], sps[j], 
                             FUN = MsCoreUtils::gnps,
                             ppm = 20)
        similarity_matrix[i, j] <- sim
        similarity_matrix[j, i] <- sim
      }
    }
  }
  
  return(similarity_matrix)
}

# Feature selection using statistical tests
feature_selection_stats <- function(features, classes) {
  p_values <- numeric(ncol(features))
  fold_changes <- numeric(ncol(features))
  
  for (i in 1:ncol(features)) {
    healthy_vals <- features[classes == "Healthy", i]
    disease_vals <- features[classes == "Disease", i]
    
    # t-test
    test_result <- t.test(disease_vals, healthy_vals)
    p_values[i] <- test_result$p.value
    
    # Fold change
    fold_changes[i] <- log2(mean(disease_vals) / mean(healthy_vals))
  }
  
  # Multiple testing correction
  adj_p_values <- p.adjust(p_values, method = "fdr")
  
  feature_stats <- data.frame(
    feature = colnames(features),
    p_value = p_values,
    adj_p_value = adj_p_values,
    log2_fc = fold_changes,
    significant = adj_p_values < 0.05
  )
  
  return(feature_stats[order(feature_stats$adj_p_value), ])
}

# Custom confusion matrix function if caret is not available
calculate_confusion_matrix <- function(predictions, actual) {
  conf_mat <- table(Predicted = predictions, Actual = actual)
  
  # Calculate metrics
  accuracy <- sum(diag(conf_mat)) / sum(conf_mat)
  
  # For binary classification
  if (nrow(conf_mat) == 2) {
    tp <- conf_mat[2, 2]
    tn <- conf_mat[1, 1]
    fp <- conf_mat[2, 1]
    fn <- conf_mat[1, 2]
    
    sensitivity <- tp / (tp + fn)
    specificity <- tn / (tn + fp)
    precision <- tp / (tp + fp)
  } else {
    sensitivity <- NA
    specificity <- NA
    precision <- NA
  }
  
  return(list(
    table = conf_mat,
    overall = c(Accuracy = accuracy, Sensitivity = sensitivity, 
                Specificity = specificity, Precision = precision)
  ))
}

# Evaluate model performance
evaluate_model <- function(model, test_x, test_y, model_name) {
  if (is.null(model)) {
    cat("Model", model_name, "is NULL, skipping evaluation\n")
    return(NULL)
  }
  
  if (model_name == "RandomForest") {
    predictions <- predict(model, test_x)
    probabilities <- predict(model, test_x, type = "prob")[, "Disease"]
  } else if (model_name == "SVM") {
    predictions <- predict(model, test_x)
    prob_matrix <- attr(predict(model, test_x, probability = TRUE), "probabilities")
    probabilities <- prob_matrix[, "Disease"]
  }
  
  # Confusion matrix
  conf_matrix <- calculate_confusion_matrix(predictions, test_y)
  
  # ROC analysis (if ROCR is available)
  if (requireNamespace("ROCR", quietly = TRUE)) {
    pred_obj <- ROCR::prediction(probabilities, test_y)
    perf_obj <- ROCR::performance(pred_obj, "tpr", "fpr")
    auc_obj <- ROCR::performance(pred_obj, "auc")
    auc_value <- auc_obj@y.values[[1]]
    
    roc_data <- data.frame(
      fpr = perf_obj@x.values[[1]],
      tpr = perf_obj@y.values[[1]]
    )
  } else {
    auc_value <- NA
    roc_data <- data.frame(fpr = numeric(0), tpr = numeric(0))
  }
  
  return(list(
    model_name = model_name,
    confusion_matrix = conf_matrix,
    predictions = predictions,
    probabilities = probabilities,
    roc_data = roc_data,
    auc = auc_value
  ))
}

# Simulate IMS-MS data
simulate_ims_data <- function(n_scans = 1000, n_drift_bins = 200, n_mz_bins = 500) {
  # Create coordinate system
  drift_times <- seq(10, 50, length.out = n_drift_bins)  # ms
  mz_values <- seq(100, 1000, length.out = n_mz_bins)
  retention_times <- seq(60, 1800, length.out = n_scans)  # seconds
  
  # Simulate IMS-MS data structure
  ims_data <- list()
  
  for (scan in 1:min(100, n_scans)) {  # Limit for memory
    # Create 2D IMS-MS spectrum (drift time x m/z)
    intensity_matrix <- matrix(0, nrow = n_drift_bins, ncol = n_mz_bins)
    
    # Add some peaks
    n_peaks <- sample(20:80, 1)
    
    for (peak in 1:n_peaks) {
      # Random peak position
      dt_center <- sample(1:n_drift_bins, 1)
      mz_center <- sample(1:n_mz_bins, 1)
      
      # Peak intensity
      peak_intensity <- rlnorm(1, meanlog = 8, sdlog = 1)
      
      # Add Gaussian peak
      for (dt in max(1, dt_center-5):min(n_drift_bins, dt_center+5)) {
        for (mz in max(1, mz_center-3):min(n_mz_bins, mz_center+3)) {
          distance <- sqrt((dt - dt_center)^2 + (mz - mz_center)^2)
          if (distance < 5) {
            intensity_matrix[dt, mz] <- intensity_matrix[dt, mz] + 
              peak_intensity * exp(-distance^2 / 4)
          }
        }
      }
    }
    
    ims_data[[scan]] <- list(
      scan_number = scan,
      retention_time = retention_times[scan],
      drift_times = drift_times,
      mz_values = mz_values,
      intensity_matrix = intensity_matrix
    )
  }
  
  return(ims_data)
}

detect_ims_peaks <- function(ims_scan, intensity_threshold = 1000) {
  intensity_matrix <- ims_scan$intensity_matrix
  drift_times <- ims_scan$drift_times
  mz_values <- ims_scan$mz_values
  
  # Find peaks above threshold
  peak_positions <- which(intensity_matrix > intensity_threshold, arr.ind = TRUE)
  
  if (nrow(peak_positions) == 0) {
    return(data.frame())
  }
  
  # Extract peak information
  peaks <- data.frame(
    drift_time_index = peak_positions[, 1],
    mz_index = peak_positions[, 2],
    drift_time = drift_times[peak_positions[, 1]],
    mz = mz_values[peak_positions[, 2]],
    intensity = intensity_matrix[peak_positions]
  )
  
  # Calculate collision cross section (CCS) - simplified relationship
  peaks$ccs <- calculate_ccs(peaks$drift_time, peaks$mz)

}

    protein_A = rlnorm(n_patients, 3, 0.5),
    protein_B = rlnorm(n_patients, 4, 0.8),
    protein_C = rlnorm(n_patients, 3.5, 0.6)
  )
  
  # Merge data
  combined_data <- merge(patients, biomarker_data, by = "patient_id")

  # Higher protein levels associated with better survival

  observed_times <- pmin(survival_times, max_followup)
  
  combined_data$time <- observed_times
  combined_data$event = event_status
  
  # Categorize biomarker levels
  combined_data$protein_A_high <- combined_data$protein_A > median(combined_data$protein_A)
  
  return(combined_data)

# Create protein-protein interaction network from correlation data
create_protein_network <- function(correlation_matrix, threshold = 0.7) {
  # Convert correlation matrix to adjacency matrix
  adj_matrix <- abs(correlation_matrix) > threshold
  diag(adj_matrix) <- FALSE
  
  # Create igraph object
  network <- graph_from_adjacency_matrix(adj_matrix, mode = "undirected")
  
  # Add edge weights
  edges <- get.edgelist(network)
  edge_weights <- numeric(nrow(edges))
  
  for (i in 1:nrow(edges)) {
    from_node <- edges[i, 1]
    to_node <- edges[i, 2]
    edge_weights[i] <- abs(correlation_matrix[from_node, to_node])
  }
  
  E(network)$weight <- edge_weights
  
  return(network)
}

      observed_response <- true_response + rnorm(1, 0, true_response * 0.05)  # 5% CV

          day = day,
          concentration = conc,
          response = observed_response,
          replicate = replicate
        ))
      }
    }
  }

  batch_size <- 20
  n_spectra <- length(sps_data)
  n_batches <- ceiling(n_spectra / batch_size)
  
  results <- list()
  
  for (i in 1:n_batches) {
    start_idx <- (i - 1) * batch_size + 1
    end_idx <- min(i * batch_size, n_spectra)
    
    # Process batch
    batch_sps <- sps_data[start_idx:end_idx]
    
    # Extract summary statistics
    results[[i]] <- data.frame(

      rt_range = paste(round(range(rtime(batch_sps)), 2), collapse = "-"),
      ms_levels = paste(unique(msLevel(batch_sps)), collapse = ","),
      mean_peaks = round(mean(lengths(mz(batch_sps))))
    )
    
    # Explicitly remove batch objects
    rm(batch_sps)

  }
  
  do.call(rbind, results)

feature_selection_stats <- function(features, classes) {
  p_values <- numeric(ncol(features))
  fold_changes <- numeric(ncol(features))
  
  for (i in 1:ncol(features)) {
    healthy_vals <- features[classes == "Healthy", i]
    disease_vals <- features[classes == "Disease", i]
    
    # t-test
    test_result <- t.test(disease_vals, healthy_vals)
    p_values[i] <- test_result$p.value
    
    # Fold change
    fold_changes[i] <- log2(mean(disease_vals) / mean(healthy_vals))
  }
  
  # Multiple testing correction
  adj_p_values <- p.adjust(p_values, method = "fdr")
  
  feature_stats <- data.frame(
    feature = colnames(features),
    p_value = p_values,
    adj_p_value = adj_p_values,
    log2_fc = fold_changes,
    significant = adj_p_values < 0.05
  )
  
  return(feature_stats[order(feature_stats$adj_p_value), ])
}

calculate_confusion_matrix <- function(predictions, actual) {
  conf_mat <- table(Predicted = predictions, Actual = actual)
  
  # Calculate metrics
  accuracy <- sum(diag(conf_mat)) / sum(conf_mat)
  
  # For binary classification
  if (nrow(conf_mat) == 2) {
    tp <- conf_mat[2, 2]
    tn <- conf_mat[1, 1]
    fp <- conf_mat[2, 1]
    fn <- conf_mat[1, 2]
    
    sensitivity <- tp / (tp + fn)
    specificity <- tn / (tn + fp)
    precision <- tp / (tp + fp)
  } else {
    sensitivity <- NA
    specificity <- NA
    precision <- NA
  }
  
  return(list(
    table = conf_mat,
    overall = c(Accuracy = accuracy, Sensitivity = sensitivity, 
                Specificity = specificity, Precision = precision)
  ))

simulate_ims_data <- function(n_scans = 1000, n_drift_bins = 200, n_mz_bins = 500) {
  # Create coordinate system
  drift_times <- seq(10, 50, length.out = n_drift_bins)  # ms
  mz_values <- seq(100, 1000, length.out = n_mz_bins)
  retention_times <- seq(60, 1800, length.out = n_scans)  # seconds
  
  # Simulate IMS-MS data structure
  ims_data <- list()
  
  for (scan in 1:min(100, n_scans)) {  # Limit for memory
    # Create 2D IMS-MS spectrum (drift time x m/z)
    intensity_matrix <- matrix(0, nrow = n_drift_bins, ncol = n_mz_bins)
    
    # Add some peaks
    n_peaks <- sample(20:80, 1)
    
    for (peak in 1:n_peaks) {
      # Random peak position
      dt_center <- sample(1:n_drift_bins, 1)
      mz_center <- sample(1:n_mz_bins, 1)
      
      # Peak intensity
      peak_intensity <- rlnorm(1, meanlog = 8, sdlog = 1)
      
      # Add Gaussian peak
      for (dt in max(1, dt_center-5):min(n_drift_bins, dt_center+5)) {
        for (mz in max(1, mz_center-3):min(n_mz_bins, mz_center+3)) {
          distance <- sqrt((dt - dt_center)^2 + (mz - mz_center)^2)
          if (distance < 5) {
            intensity_matrix[dt, mz] <- intensity_matrix[dt, mz] + 
              peak_intensity * exp(-distance^2 / 4)
          }
        }
      }
    }
    
    ims_data[[scan]] <- list(
      scan_number = scan,
      retention_time = retention_times[scan],
      drift_times = drift_times,
      mz_values = mz_values,
      intensity_matrix = intensity_matrix
    )
  }
  
  return(ims_data)
}

                     0.3 * (combined_data$stage == "IV") +
                     0.2 * (combined_data$age > 70))
  
  # Generate survival times using exponential distribution
  survival_times <- rexp(n_patients, rate = hazard_ratio * 0.1)
  
  # Generate censoring (assume some patients are still alive at end of study)
  max_followup <- 60  # months
  event_status <- ifelse(survival_times < max_followup, 1, 0)
  observed_times <- pmin(survival_times, max_followup)
  
  combined_data$time <- observed_times
  combined_data$event = event_status
  
  # Categorize biomarker levels
  combined_data$protein_A_high <- combined_data$protein_A > median(combined_data$protein_A)
  
  return(combined_data)
}

create_protein_network <- function(correlation_matrix, threshold = 0.7) {
  # Convert correlation matrix to adjacency matrix
  adj_matrix <- abs(correlation_matrix) > threshold
  diag(adj_matrix) <- FALSE
  
  # Create igraph object
  network <- graph_from_adjacency_matrix(adj_matrix, mode = "undirected")
  
  # Add edge weights
  edges <- get.edgelist(network)
  edge_weights <- numeric(nrow(edges))
  
  for (i in 1:nrow(edges)) {
    from_node <- edges[i, 1]
    to_node <- edges[i, 2]
    edge_weights[i] <- abs(correlation_matrix[from_node, to_node])
  }
  
  E(network)$weight <- edge_weights
  
  return(network)
}

simulate_validation_data <- function() {
  # Calibration standards
  concentrations <- c(0, 0.1, 0.5, 1, 5, 10, 50, 100)  # ng/mL
  
  validation_data <- list()
  
  # Linearity assessment
  linearity_data <- data.frame()
  for (conc in concentrations) {
    for (replicate in 1:5) {
      # Simulate instrument response with some noise
      true_response <- 1000 * conc + 500  # Linear relationship
      observed_response <- true_response + rnorm(1, 0, true_response * 0.05)  # 5% CV
      
      linearity_data <- rbind(linearity_data, data.frame(
        concentration = conc,
        response = observed_response,
        replicate = replicate
      ))
    }
  }
  
  # Precision (repeatability and intermediate precision)
  precision_data <- data.frame()
  test_concentrations <- c(1, 10, 50)  # Low, medium, high
  
  for (day in 1:3) {
    for (conc in test_concentrations) {
      for (replicate in 1:6) {
        true_response <- 1000 * conc + 500
        day_effect <- rnorm(1, 0, true_response * 0.02)  # Day-to-day variation
        observed_response <- true_response + day_effect + 
                           rnorm(1, 0, true_response * 0.03)  # Within-day variation
        
        precision_data <- rbind(precision_data, data.frame(
          day = day,
          concentration = conc,
          response = observed_response,
          replicate = replicate
        ))
      }
    }
  }
  
  # Accuracy (spike recovery)
  accuracy_data <- data.frame()
  spike_levels <- c(0.5, 5, 50)  # ng/mL
  
  for (spike in spike_levels) {
    for (replicate in 1:5) {
      expected_response <- 1000 * spike + 500
      recovery_rate <- runif(1, 0.85, 1.15)  # 85-115% recovery
      observed_response <- expected_response * recovery_rate
      
      accuracy_data <- rbind(accuracy_data, data.frame(
        spiked_concentration = spike,
        observed_response = observed_response,
        expected_response = expected_response,
        recovery_percent = recovery_rate * 100,
        replicate = replicate
      ))
    }
  }
  
  return(list(
    linearity = linearity_data,
    precision = precision_data,
    accuracy = accuracy_data
  ))
}
